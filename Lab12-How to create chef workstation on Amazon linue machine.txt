# Cookbook:: apache-book
# Recipe:: apache-recipe

# Update the system packages

  command 'yum update -y'
  action :run
end

# Install Apache (httpd)
package 'httpd' do
  action :install
end

# Enable and start the Apache service

  action [:enable, :start]
